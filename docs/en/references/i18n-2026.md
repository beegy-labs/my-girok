# Internationalization (i18n) Best Practices - 2026

This guide covers internationalization best practices as of 2026, focusing on react-i18next, localization patterns, and RTL support.

## Overview

| Feature    | Description                |
| ---------- | -------------------------- |
| Library    | react-i18next + i18next    |
| Backend    | i18next-http-backend       |
| Detection  | Browser language detection |
| Namespaces | Feature-based splitting    |

## Library Setup

### Installation

```bash
pnpm add react-i18next i18next i18next-http-backend i18next-browser-languagedetector
```

### Configuration

```typescript
// src/i18n/config.ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import HttpBackend from 'i18next-http-backend';
import LanguageDetector from 'i18next-browser-languagedetector';

i18n
  .use(HttpBackend)
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    fallbackLng: 'en',
    supportedLngs: ['en', 'ko', 'ja', 'zh'],

    // Namespaces
    ns: ['common', 'auth', 'dashboard', 'errors'],
    defaultNS: 'common',

    // Backend configuration
    backend: {
      loadPath: '/locales/{{lng}}/{{ns}}.json',
    },

    // Detection configuration
    detection: {
      order: ['querystring', 'cookie', 'localStorage', 'navigator'],
      caches: ['localStorage', 'cookie'],
    },

    // React specific
    react: {
      useSuspense: true,
    },

    // Interpolation
    interpolation: {
      escapeValue: false, // React already escapes
    },
  });

export default i18n;
```

## Usage Patterns

### useTranslation Hook

```tsx
import { useTranslation } from 'react-i18next';

function Dashboard() {
  const { t } = useTranslation('dashboard');

  return (
    <div>
      <h1>{t('title')}</h1>
      <p>{t('description')}</p>
      <button>{t('common:actions.save')}</button>
    </div>
  );
}
```

### Trans Component (JSX in Translations)

```tsx
import { Trans } from 'react-i18next';

// Translation: "Welcome, <strong>{{name}}</strong>! You have <link>{{count}} messages</link>."
function Welcome({ name, count }: { name: string; count: number }) {
  return (
    <Trans
      i18nKey="welcome"
      values={{ name, count }}
      components={{
        strong: <strong />,
        link: <a href="/messages" />,
      }}
    />
  );
}
```

## Date & Number Formatting

### Intl.DateTimeFormat

```tsx
function FormattedDate({ date }: { date: Date }) {
  const { i18n } = useTranslation();

  const formatted = new Intl.DateTimeFormat(i18n.language, {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
  }).format(date);

  return <time dateTime={date.toISOString()}>{formatted}</time>;
}
```

### Intl.NumberFormat

```tsx
function Currency({ amount }: { amount: number }) {
  const { i18n } = useTranslation();

  const currencyMap: Record<string, string> = {
    en: 'USD',
    ko: 'KRW',
    ja: 'JPY',
  };

  const formatted = new Intl.NumberFormat(i18n.language, {
    style: 'currency',
    currency: currencyMap[i18n.language] || 'USD',
  }).format(amount);

  return <span>{formatted}</span>;
}
```

## RTL Support

### Language Direction Detection

```tsx
import { useTranslation } from 'react-i18next';

function App() {
  const { i18n } = useTranslation();

  // Automatic RTL detection
  const dir = i18n.dir(); // 'ltr' or 'rtl'

  return (
    <html lang={i18n.language} dir={dir}>
      <body className={dir === 'rtl' ? 'rtl' : 'ltr'}>{/* App content */}</body>
    </html>
  );
}
```

### RTL-Aware CSS

```css
/* Base styles (LTR) */
.sidebar {
  margin-left: 1rem;
  text-align: left;
}

/* RTL overrides */
[dir='rtl'] .sidebar {
  margin-left: 0;
  margin-right: 1rem;
  text-align: right;
}

/* Or use logical properties (recommended) */
.sidebar {
  margin-inline-start: 1rem;
  text-align: start;
}
```

### Tailwind RTL

```html
<!-- Using Tailwind's RTL utilities -->
<div class="ms-4 me-2">
  <!-- ms = margin-start, me = margin-end -->
  <!-- Works correctly in both LTR and RTL -->
</div>

<div class="text-start">
  <!-- Aligns to start (left in LTR, right in RTL) -->
</div>
```

function LanguageSwitcher() {
  const { i18n } = useTranslation();

  const changeLanguage = (lng: string) => {
    i18n.changeLanguage(lng);
  };

  return (
    <select value={i18n.language} onChange={(e) => changeLanguage(e.target.value)}>

## Lazy Loading Namespaces

```tsx
import { Suspense, lazy } from 'react';
import { useTranslation } from 'react-i18next';

// Load namespace on demand
function Settings() {
  const { t, ready } = useTranslation('settings', { useSuspense: false });

  if (!ready) return <div>Loading...</div>;

  return <div>{t('title')}</div>;
}

// Or with Suspense
const LazySettings = lazy(() => import('./Settings'));

function App() {
  return (
    <Suspense fallback={<div>Loading...</div>}>
      <LazySettings />
    </Suspense>
  );
}
```

## Type Safety

### Typed Translation Keys

```typescript
// src/types/i18n.d.ts
import 'i18next';
import common from '../locales/en/common.json';
import auth from '../locales/en/auth.json';

declare module 'i18next' {
  interface CustomTypeOptions {
    defaultNS: 'common';
    resources: {
      common: typeof common;
      auth: typeof auth;
    };
  }
}
```

## Best Practices Summary

| Practice             | Description                              |
| -------------------- | ---------------------------------------- |
| Namespace by feature | `auth.json`, `dashboard.json`            |
| Use nested keys      | `nav.home`, `actions.save`               |
| Avoid concatenation  | Use interpolation `{{name}}`             |
| Test all languages   | Check layout in RTL languages            |
| Use logical CSS      | `margin-inline-start` over `margin-left` |

## Anti-Patterns to Avoid

| Don't               | Do                          | Reason          |
| ------------------- | --------------------------- | --------------- |
| Concatenate strings | Use interpolation           | Grammar differs |
| Hard-code dates     | Use Intl.DateTimeFormat     | Format varies   |
| Ignore RTL          | Test Arabic/Hebrew          | 10%+ of users   |
| Single huge file    | Split by namespace          | Performance     |
| Skip context        | Provide translators context | Accuracy        |
